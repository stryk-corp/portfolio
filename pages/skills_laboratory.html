api-service.js 
// RESTful API with Express.js and MongoDB
const express = require('express');
const mongoose = require('mongoose');
const cors = require('cors');

class APIService {
  constructor() {
    this.app = express();
    this.setupMiddleware();
    this.setupRoutes();
  }

  setupMiddleware() {
    this.app.use(cors());
    this.app.use(express.json());
    this.app.use(express.urlencoded({ extended: true }));
  }

  setupRoutes() {
    // User management endpoints
    this.app.get('/api/users', this.getUsers);
    this.app.post('/api/users', this.createUser);
    this.app.put('/api/users/:id', this.updateUser);
    this.app.delete('/api/users/:id', this.deleteUser);
  }

  async getUsers(req, res) {
    try {
      const users = await User.find()
        .select('-password')
        .sort({ createdAt: -1 });
      
      res.json({
        success: true,
        data: users,
        count: users.length
      });
    } catch (error) {
      res.status(500).json({
        success: false,
        message: error.message
      });
    }
  }

  start(port = 3000) {
    this.app.listen(port, () => {
      console.log(`API Server running on port ${port}`);
    });
  }
}

module.exports = APIService;

RESTful API Development

This Express.js API demonstrates modern web service architecture with:

  • Modular Design: Class-based structure for maintainability
  • Error Handling: Comprehensive try-catch with meaningful responses
  • Security: CORS configuration and input validation
  • Database Integration: MongoDB with Mongoose ODM
  • RESTful Conventions: Standard HTTP methods and status codes
