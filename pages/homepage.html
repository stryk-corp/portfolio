// Bridging Web and Desktop APIs
const { ipcRenderer } = require('electron');
const express = require('express');

class CrossPlatformBridge {
  constructor() {
    this.webServer = express();
    this.setupRoutes();
  }

  // Desktop to Web communication
  async sendToWeb(data) {
    return await ipcRenderer.invoke('web-api', data);
  }

  // Web to Desktop communication  
  setupRoutes() {
    this.webServer.post('/desktop-action', (req, res) => {
      ipcRenderer.send('desktop-action', req.body);
      res.json({ status: 'success' });
    });
  }
}
